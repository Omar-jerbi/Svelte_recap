"pulisci i file prefatti di svelte"

add in file rollup.config.js
	import sveltePreprocess  from 'svelte-preprocess';
	(sotto svelte -> compilerOptions) 
			, 
			preprocess: [
				sveltePreprocess({
					sourceMap: !production,
					postcss: {
						plugins: [require("autoprefixer")],
					},
				}),
			],
		}),
